المصفوفات (Arrays)

ما هي المصفوفة؟

المصفوفة هي هيكل بيانات خطي يخزن مجموعة من العناصر من نفس النوع في مواضع ذاكرة متتالية.

التمثيل في الذاكرة:

5
12
8
3
16


المؤشر:    0      1      2      3      4

// تصريح مصفوفة بحجم 5 عناصر int numbers[5] = {5, 12, 8, 3, 16}; // الوصول للعناصر cout << numbers[0]; // النتيجة: 5 cout << numbers[2]; // النتيجة: 8
ملاحظة: ترقيم عناصر المصفوفة يبدأ من الصفر (0) وليس الواحد.

خصائص المصفوفات

المزايا:

  • الوصول السريع: O(1) للوصول لأي عنصر بالمؤشر
  • استخدام الذاكرة: عناصر متتالية في الذاكرة
  • البساطة: سهولة في الفهم والتطبيق

العيوب:

  • الحجم الثابت: لا يمكن تغيير الحجم بعد التصريح
  • الإدراج والحذف: عمليات مكلفة O(n)
  • إهدار الذاكرة: إذا لم تُستخدم كل العناصر
// مثال على مشكلة الحجم الثابت int students[100]; // مصفوفة لـ 100 طالب // ماذا لو احتجنا لـ 150 طالب؟ // ماذا لو استخدمنا 20 طالب فقط؟

العمليات الأساسية على المصفوفات

void insertAt(int arr[], int &size, int index, int value) { // إزاحة العناصر لليمين for(int i = size; i > index; i--) { arr[i] = arr[i-1]; } // إدراج القيمة الجديدة arr[index] = value; size++; cout << "تم إدراج " << value << " في الموضع " << index << endl; }

البحث في المصفوفات

البحث الخطي (Linear Search)

int linearSearch(int arr[], int size, int target) { for(int i = 0; i < size; i++) { if(arr[i] == target) { return i; // إرجاع موضع العنصر } } return -1; // العنصر غير موجود } // استخدام الدالة int numbers[] = {5, 12, 8, 3, 16}; int position = linearSearch(numbers, 5, 8); if(position != -1) { cout << "تم العثور على العنصر في الموضع: " << position << endl; } else { cout << "العنصر غير موجود" << endl; }
التعقيد الزمني:
  • أفضل حالة: O(1) - العنصر في بداية المصفوفة
  • أسوأ حالة: O(n) - العنصر في نهاية المصفوفة أو غير موجود
  • متوسط: O(n/2)

تطبيق عملي: نظام إدارة الدرجات

#include <iostream> using namespace std; class GradeManager { private: int grades[50]; // مصفوفة لحفظ الدرجات int count; // عدد الدرجات الحالي public: GradeManager() : count(0) {} // إضافة درجة جديدة void addGrade(int grade) { if(count < 50) { grades[count] = grade; count++; cout << "تم إضافة الدرجة: " << grade << endl; } else { cout << "لا يمكن إضافة المزيد من الدرجات!" << endl; } } // حساب المتوسط double calculateAverage() { if(count == 0) return 0; int sum = 0; for(int i = 0; i < count; i++) { sum += grades[i]; } return (double)sum / count; } // عرض جميع الدرجات void displayGrades() { cout << "الدرجات: "; for(int i = 0; i < count; i++) { cout << grades[i] << " "; } cout << endl; cout << "المتوسط: " << calculateAverage() << endl; } };

القوائم المترابطة (Linked Lists)

مقدمة القوائم المترابطة

القائمة المترابطة هي هيكل بيانات ديناميكي يتكون من سلسلة من العقد (Nodes)، كل عقدة تحتوي على بيانات ومؤشر للعقدة التالية.

بنية العقدة (Node):

┌─────────┬─────────┐
│  البيانات │  المؤشر  │
│  (Data) │ (Next)  │
└─────────┴─────────┘
// تعريف العقدة class Node { public: int data; // البيانات Node* next; // مؤشر للعقدة التالية // منشئ العقدة Node(int value) { data = value; next = nullptr; } };

مثال قائمة مترابطة:

Head → [5|→] → [12|→] → [8|→] → [3|NULL]

إنشاء وإدارة القائمة المترابطة

class LinkedList { private: Node* head; // مؤشر لبداية القائمة public: // منشئ القائمة LinkedList() { head = nullptr; cout << "تم إنشاء قائمة مترابطة فارغة" << endl; } // فحص إذا كانت القائمة فارغة bool isEmpty() { return head == nullptr; } // حساب عدد العقد int size() { int count = 0; Node* current = head; while(current != nullptr) { count++; current = current->next; } return count; } };
مزايا القوائم المترابطة:
  • حجم ديناميكي - يمكن الزيادة والنقصان حسب الحاجة
  • إدراج وحذف سريع في البداية O(1)
  • عدم إهدار الذاكرة

عمليات الإدراج

1. الإدراج في البداية

void insertAtBeginning(int value) { Node* newNode = new Node(value); newNode->next = head; head = newNode; cout << "تم إضافة " << value << " في بداية القائمة" << endl; }

خطوات الإدراج في البداية:

1. إنشاء عقدة جديدة

2. ربط العقدة الجديدة بـ head الحالي

3. تحديث head ليشير للعقدة الجديدة


قبل: Head → [5|→] → [12|NULL]
بعد: Head → [3|→] → [5|→] → [12|NULL]

2. الإدراج في النهاية

void insertAtEnd(int value) { Node* newNode = new Node(value); if(head == nullptr) { head = newNode; // القائمة فارغة return; } // البحث عن آخر عقدة Node* current = head; while(current->next != nullptr) { current = current->next; } current->next = newNode; cout << "تم إضافة " << value << " في نهاية القائمة" << endl; }

عمليات الحذف

حذف عقدة بقيمة محددة

bool deleteNode(int value) { // فحص إذا كانت القائمة فارغة if(head == nullptr) { cout << "القائمة فارغة" << endl; return false; } // حذف العقدة الأولى if(head->data == value) { Node* nodeToDelete = head; head = head->next; delete nodeToDelete; cout << "تم حذف " << value << " من بداية القائمة" << endl; return true; } // البحث والحذف Node* current = head; while(current->next != nullptr && current->next->data != value) { current = current->next; } if(current->next == nullptr) { cout << "العنصر " << value << " غير موجود" << endl; return false; } Node* nodeToDelete = current->next; current->next = nodeToDelete->next; delete nodeToDelete; cout << "تم حذف " << value << " من القائمة" << endl; return true; }

مثال على الحذف:

قبل الحذف: [5|→] → [12|→] → [8|→] → [3|NULL]
حذف 12: [5|→] → [8|→] → [3|NULL]

تطبيق عملي: قائمة جهات الاتصال

#include <iostream> #include <string> using namespace std; struct Contact { string name; string phone; Contact* next; Contact(string n, string p) : name(n), phone(p), next(nullptr) {} }; class ContactList { private: Contact* head; public: ContactList() : head(nullptr) {} void addContact(string name, string phone) { Contact* newContact = new Contact(name, phone); newContact->next = head; head = newContact; cout << "تم إضافة جهة الاتصال: " << name << endl; } void displayContacts() { if(head == nullptr) { cout << "قائمة الاتصالات فارغة" << endl; return; } cout << "قائمة جهات الاتصال:" << endl; Contact* current = head; int index = 1; while(current != nullptr) { cout << index << ". " << current->name << " - " << current->phone << endl; current = current->next; index++; } } bool deleteContact(string name) { if(head == nullptr) return false; // حذف من البداية if(head->name == name) { Contact* temp = head; head = head->next; delete temp; return true; } // البحث والحذف Contact* current = head; while(current->next != nullptr && current->next->name != name) { current = current->next; } if(current->next == nullptr) return false; Contact* temp = current->next; current->next = temp->next; delete temp; return true; } };

القوائم المضاعفة والدائرية

القوائم المترابطة المضاعفة (Doubly Linked Lists)

قائمة مترابطة حيث كل عقدة لها مؤشرين: واحد للعقدة التالية وآخر للعقدة السابقة.

بنية العقدة المضاعفة:

┌─────────┬─────────┬─────────┐
│  السابق  │  البيانات │  التالي  │
│ (Prev)  │ (Data)  │ (Next)  │
└─────────┴─────────┴─────────┘
class DoublyNode { public: int data; DoublyNode* next; // مؤشر للعقدة التالية DoublyNode* prev; // مؤشر للعقدة السابقة DoublyNode(int value) { data = value; next = nullptr; prev = nullptr; } }; class DoublyLinkedList { private: DoublyNode* head; // بداية القائمة DoublyNode* tail; // نهاية القائمة public: DoublyLinkedList() { head = nullptr; tail = nullptr; } };

مثال قائمة مضاعفة:

NULL ← [5] ↔ [12] ↔ [8] ↔ [3] → NULL

عمليات القوائم المضاعفة

الإدراج في البداية

void insertAtBeginning(int value) { DoublyNode* newNode = new DoublyNode(value); if(head == nullptr) { head = tail = newNode; // القائمة فارغة } else { newNode->next = head; head->prev = newNode; head = newNode; } cout << "تم إضافة " << value << " في بداية القائمة المضاعفة" << endl; }

الإدراج في النهاية

void insertAtEnd(int value) { DoublyNode* newNode = new DoublyNode(value); if(tail == nullptr) { head = tail = newNode; // القائمة فارغة } else { tail->next = newNode; newNode->prev = tail; tail = newNode; } cout << "تم إضافة " << value << " في نهاية القائمة المضاعفة" << endl; }
مزايا القوائم المضاعفة:
  • إمكانية التنقل في الاتجاهين
  • حذف سريع عند معرفة موقع العقدة
  • تنفيذ عمليات معقدة بسهولة أكبر

القوائم المترابطة الدائرية (Circular Linked Lists)

قائمة مترابطة حيث العقدة الأخيرة تشير إلى العقدة الأولى، مكونة دائرة مغلقة.

القائمة الدائرية:

    [5] → [12]
     ↑      ↓
    [3] ← [8]
class CircularLinkedList { private: Node* head; public: CircularLinkedList() : head(nullptr) {} void insertAtEnd(int value) { Node* newNode = new Node(value); if(head == nullptr) { head = newNode; newNode->next = head; // تشير لنفسها } else { Node* current = head; // البحث عن آخر عقدة while(current->next != head) { current = current->next; } current->next = newNode; newNode->next = head; // ربط بالبداية } cout << "تم إضافة " << value << " في القائمة الدائرية" << endl; } };

عرض القوائم الدائرية

void display() { if(head == nullptr) { cout << "القائمة الدائرية فارغة" << endl; return; } cout << "عناصر القائمة الدائرية: "; Node* current = head; // استخدام do-while للتأكد من عرض العقدة الأولى do { cout << current->data << " "; current = current->next; } while(current != head); // التوقف عند العودة للبداية cout << "(عودة للبداية)" << endl; } // البحث في القائمة الدائرية bool search(int value) { if(head == nullptr) return false; Node* current = head; do { if(current->data == value) { cout << "تم العثور على " << value << " في القائمة" << endl; return true; } current = current->next; } while(current != head); cout << "لم يتم العثور على " << value << endl; return false; }
تحذير هام: في القوائم الدائرية، تأكد من استخدام شروط توقف صحيحة لتجنب الحلقات اللانهائية!

المقارنة بين أنواع القوائم

القائمة البسيطة

  • بساطة في التطبيق
  • ذاكرة أقل لكل عقدة
  • اتجاه واحد فقط

القائمة المضاعفة

  • تنقل في الاتجاهين
  • حذف أسهل
  • ذاكرة إضافية

القائمة الدائرية

  • لا نهاية للقائمة
  • مثالية للتطبيقات الدائرية
  • حذر من الحلقات اللانهائية

تطبيقات عملية:

// مثال: قائمة تشغيل الموسيقى (دائرية) class Playlist { Node* currentSong; void nextSong() { if(currentSong != nullptr) { currentSong = currentSong->next; cout << "الأغنية التالية: " << currentSong->data << endl; } } void previousSong() { // في القائمة المضاعفة الدائرية if(currentSong != nullptr) { currentSong = currentSong->prev; cout << "الأغنية السابقة: " << currentSong->data << endl; } } }; // مثال: نظام تنقل المتصفح (مضاعف) class BrowserHistory { DoublyNode* currentPage; void goBack() { if(currentPage && currentPage->prev) { currentPage = currentPage->prev; cout << "العودة للصفحة: " << currentPage->data << endl; } } void goForward() { if(currentPage && currentPage->next) { currentPage = currentPage->next; cout << "التقدم للصفحة: " << currentPage->data << endl; } } };

المكدس (Stack)

ما هو المكدس؟

المكدس هو هيكل بيانات خطي يتبع مبدأ LIFO (Last In, First Out) - آخر داخل أول خارج.

تشبيه المكدس:

4 (Top)

3

2

1 (Bottom)

مثل كومة الأطباق - نضع ونأخذ من الأعلى فقط!

العمليات الأساسية:
  • Push: إضافة عنصر لأعلى المكدس
  • Pop: إزالة العنصر العلوي
  • Peek/Top: النظر للعنصر العلوي دون إزالته
  • isEmpty: فحص إذا كان المكدس فارغاً

تطبيق المكدس بالمصفوفة

class ArrayStack { private: int* data; // مصفوفة لحفظ العناصر int top; // مؤشر لأعلى عنصر int capacity; // السعة القصوى public: // منشئ المكدس ArrayStack(int size) { capacity = size; data = new int[capacity]; top = -1; // المكدس فارغ cout << "تم إنشاء مكدس بسعة " << size << endl; } // مدمر المكدس ~ArrayStack() { delete[] data; cout << "تم تحرير ذاكرة المكدس" << endl; } // فحص إذا كان المكدس فارغاً bool isEmpty() { return top == -1; } // فحص إذا كان المكدس ممتلئاً bool isFull() { return top == capacity - 1; } };

عمليات Push و Pop

عملية Push (الإضافة)

void push(int value) { // فحص إذا كان المكدس ممتلئاً if(isFull()) { cout << "خطأ: المكدس ممتلئ!" << endl; return; } // إضافة العنصر وتحديث المؤشر data[++top] = value; cout << "تم دفع " << value << " للمكدس" << endl; }

عملية Pop (الإزالة)

int pop() { // فحص إذا كان المكدس فارغاً if(isEmpty()) { cout << "خطأ: المكدس فارغ!" << endl; return -1; // قيمة خطأ } // إرجاع العنصر وتقليل المؤشر int value = data[top--]; cout << "تم سحب " << value << " من المكدس" << endl; return value; }

عملية Peek (النظر)

int peek() { if(isEmpty()) { cout << "المكدس فارغ" << endl; return -1; } return data[top]; // إرجاع دون تغيير المؤشر }

مثال تطبيقي:

push(10) → top = 0, المكدس: [10]
push(20) → top = 1, المكدس: [10, 20]
push(30) → top = 2, المكدس: [10, 20, 30]
pop() → يرجع 30, top = 1, المكدس: [10, 20]
peek() → يرجع 20, المكدس: [10, 20] (بدون تغيير)

تطبيق عملي: فحص توازن الأقواس

واحد من أشهر تطبيقات المكدس هو فحص توازن الأقواس في التعابير الرياضية والبرمجية.

bool isBalanced(string expression) { ArrayStack bracketStack(100); for(int i = 0; i < expression.length(); i++) { char ch = expression[i]; // دفع الأقواس المفتوحة if(ch == '(' || ch == '[' || ch == '{') { bracketStack.push(ch); } // فحص الأقواس المغلقة else if(ch == ')' || ch == ']' || ch == '}') { if(bracketStack.isEmpty()) { return false; // قوس مغلق بدون مفتوح } char top = bracketStack.pop(); // فحص التطابق if((ch == ')' && top != '(') || (ch == ']' && top != '[') || (ch == '}' && top != '{')) { return false; // أقواس غير متطابقة } } } // يجب أن يكون المكدس فارغاً في النهاية return bracketStack.isEmpty(); }

أمثلة على الاختبار:

"([{}])" → متوازن ✓
"([)]" → غير متوازن ✗
"((()))" → متوازن ✓
"({[}])" → غير متوازن ✗
"" → متوازن ✓

تطبيقات أخرى للمكدس

1. تقييم التعابير الرياضية

// تحويل التعبير من Infix إلى Postfix // مثال: (3 + 4) * 5 → 3 4 + 5 * string infixToPostfix(string infix) { ArrayStack operatorStack(100); string postfix = ""; for(char ch : infix) { if(isdigit(ch)) { postfix += ch; postfix += " "; } else if(ch == '(') { operatorStack.push(ch); } else if(ch == ')') { while(!operatorStack.isEmpty() && operatorStack.peek() != '(') { postfix += operatorStack.pop(); postfix += " "; } operatorStack.pop(); // إزالة '(' } // معالجة العمليات الأخرى... } return postfix; }

2. تتبع استدعاءات الدوال (Function Call Stack)

// مثال: حساب المضروب باستخدام الاستدعاء المتكرر int factorial(int n) { if(n <= 1) { return 1; // الحالة الأساسية } // كل استدعاء يُحفظ في المكدس return n * factorial(n - 1); } // عند استدعاء factorial(5): // المكدس: factorial(5) ← factorial(4) ← factorial(3) ← factorial(2) ← factorial(1)

3. عمليات Undo/Redo

class TextEditor { private: ArrayStack undoStack; ArrayStack redoStack; string currentText; public: void typeText(string text) { undoStack.push(currentText); // حفظ الحالة الحالية currentText += text; // مسح redo stack عند كتابة جديدة while(!redoStack.isEmpty()) { redoStack.pop(); } } void undo() { if(!undoStack.isEmpty()) { redoStack.push(currentText); currentText = undoStack.pop(); } } void redo() { if(!redoStack.isEmpty()) { undoStack.push(currentText); currentText = redoStack.pop(); } } };
